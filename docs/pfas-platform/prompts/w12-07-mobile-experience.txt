@media (max-width: 768px) {
  button,
  a,
  [role="button"],
  input[type="checkbox"],

    min-height: 44px;
    min-width: 44px;

    padding: 12px;

  select,

    font-size: 16px;

  /* Safe area insets for notched phones */
  .mobile-nav,
  .mobile-filter-sheet {

  /* Disable hover effects on touch */
  @media (hover: none) {
    .product-card:hover {
      transform: none;

  }

  .mobile-nav {
    max-height: 100vh;
    overflow-y: auto;
