## Usage

### Import patterns directly:
```tsx
import { HeroCenteredGradient } from "@/lib/patterns/patterns/heroes";
import { FeaturesIconGrid } from "@/lib/patterns/patterns/features";
```

### Use SectionRenderer:
```tsx
import { SectionRenderer } from "@/components/preview/SectionRenderer";

<SectionRenderer
  section={{ id: "hero-1", patternId: "hero-centered-gradient", props: {...} }}
  branding={brandingConfig}
/>
```
