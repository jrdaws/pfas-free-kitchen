### 3. Pattern Card Component

Show pattern metadata before rendering:

```tsx
function PatternCard({
  patternId,
  name,
  variants,
  children,
}: {
  patternId: string;
  name: string;
  variants?: string[];
  children: React.ReactNode;
}) {
  return (
    <div className="border-b border-white/5">
      {/* Pattern info header */}
      <div className="max-w-7xl mx-auto px-6 py-4 flex items-center justify-between bg-white/[0.02]">
        <div>
          <code className="text-xs text-orange-400 font-mono">{patternId}</code>
          <span className="text-white/60 text-sm ml-4">{name}</span>
        </div>
        {variants && (
          <div className="flex gap-2">
            {variants.map(v => (
              <span key={v} className="text-xs px-2 py-1 bg-white/5 rounded text-white/40">
                {v}
              </span>
            ))}
          </div>
        )}
      </div>
      
      {/* Pattern render */}
      {children}
    </div>
  );
}
```
